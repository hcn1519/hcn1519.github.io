// IntPoint.m
#import "IntPoint.h"

@implementation IntPoint

- (void)print {
    NSLog(@"Point x: %d, y: %d", self.x, self.y);
}

- (void)set:(int)x and:(int)y {
    self.x = x;
    self.y = y;
}

- (IntPoint *)add:(IntPoint *)point {
    IntPoint *result = [[IntPoint alloc] init];

    result.x = self.x + point.x;
    result.y = self.y + point.y;
    return result;
}
@end
